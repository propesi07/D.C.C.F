    <script>
        function checkPassword() {
            const input = document.getElementById('password-input').value;
            if (input === 'dccf') {
                document.getElementById('login-screen').classList.add('hidden');
                document.getElementById('main-content').classList.remove('hidden');
                loadPosts();
            } else {
                alert('Senha incorreta. Acesso negado.');
            }
        }

        function loadPosts() {
            const container = document.getElementById('posts-container');
            container.innerHTML = '';
            const posts = JSON.parse(localStorage.getItem('dccf_posts')) || [];
            
            posts.forEach((post, index) => {
                const postDiv = document.createElement('div');
                postDiv.className = 'post';
                postDiv.innerHTML = `
                    <strong style="color: #ecc94b;">${post.title}</strong> - <small>${post.date}</small><br>
                    <small style="color: #94a3b8;">Agentes: ${post.agents}</small><br><br>
                    <div>${post.content}</div>
                    <button class="delete-btn" onclick="deletePost(${index})">Apagar</button>
                `;
                container.appendChild(postDiv);
            });
        }

        function addPost() {
            const title = document.getElementById('title-input').value;
            const agents = document.getElementById('agents-input').value;
            const date = document.getElementById('date-input').value;
            const content = document.getElementById('content-input').value;

            if (!title || !agents || !content) return alert('Preencha a operação, agentes e o relatório!');

            const posts = JSON.parse(localStorage.getItem('dccf_posts')) || [];
            posts.push({ title, agents, date, content });
            localStorage.setItem('dccf_posts', JSON.stringify(posts));

            document.getElementById('title-input').value = '';
            document.getElementById('agents-input').value = '';
            document.getElementById('date-input').value = '';
            document.getElementById('content-input').value = '';
            loadPosts();
        }

        function deletePost(index) {
            if(confirm("Tem certeza que deseja apagar este arquivo?")) {
                const posts = JSON.parse(localStorage.getItem('dccf_posts')) || [];
                posts.splice(index, 1);
                localStorage.setItem('dccf_posts', JSON.stringify(posts));
                loadPosts();
            }
        }
    </script>
